\begin{algorithm}[H]
\caption{Detect hashtags and create conversational graphs}

\begin{algorithmic}[1]
\SetAlgoLined
\KwData{Microblog messages}
\KwResult{Hashtags user graphs}

initialization\;
% \KwIn{Integers $a \geq 0$ and $b \geq 0$}
\Procedure{ArraySum}{$A$}

\ForEach{messages in the corpus}{
    read message \;

    \If{words that begins with \# and ends with \# (Sina Weibo hashtag pattern)}{
        hashtags in message\;

        \If{quotes or retweet in the message}{
            add a directed edge to the graph user A -> user B \;
        }{
            next message \;
        }
    }{
        next message \;
    }
}
\EndProcedure


\end{algorithmic}
\end{algorithm}
